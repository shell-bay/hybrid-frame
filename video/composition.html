Simple Python API
demo.py
from hybrid_frame import HybridFrame

# Wrap Pandas DataFrame with zero copy
hf = HybridFrame.from_pandas(df)

# Lazy DuckDB execution pipeline
result = (
  hf
  .filter("age > 25")
  .sort_values("salary", ascending=False)
  .head(10)
)

# Direct export to scikit-learn format
X, y = result.to_ml_ready("salary")
