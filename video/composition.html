Sort + Head operations up to 26x faster

Simple API

from hybrid_frame import HybridFrame

# Create from Pandas
hf = HybridFrame.from_pandas(df)

# Lazy DuckDB operations
result = (
  hf
  .filter("age > 25")
  .sort_values("salary", ascending=False)
  .head(10)
)

# Materialize for ML
X, y = result.to_ml_ready("salary")
